    "# Create a function to transform initial df to standardize both\n",
    "def fxTransformBases(df):\n",
    "    return (\n",
    "        df.withColumn(\"Date\", f.to_date(f.to_timestamp(f.col(\"Start\"), \"yyyy-MM-dd'T'HH:mm:ss.SSS'Z'\"), \"yyyy-MM-dd\"))\n",
    "        .select(\"Queue\", \"Date\", \"Type\")\n",
    "        .distinct()\n",
    "        .withColumn(\"Type\", f.upper(f.col(\"Type\")))\n",
    "    )\n",
    "\n",
    "\n",
    "# Create a function to apply naming convention to df\n",
    "def fxApplyNamingConvention(df, namingconvention_df):\n",
    "\n",
    "    # Join df to get a new df with naming convention applied\n",
    "    joined_df = None\n",
    "    joined_df = df.join(\n",
    "        namingconvention_df,\n",
    "        df[\"Queue\"] == namingconvention_df[\"Description\"],\n",
    "        how=\"left\"\n",
    "    )\n",
    "\n",
    "    # Replace values on 'Queue' column basis n convention\n",
    "    replaced_df = joined_df.withColumn(\n",
    "        \"Queue\", f.coalesce(f.col(\"Code\"), f.col(\"Queue\"))\n",
    "    )\n",
    "\n",
    "    # Remove as colunas temporárias usadas para o mapeamento\n",
    "    nameconv_df = replaced_df.drop(\"Description\", \"Code\")\n",
    "\n",
    "    # Remove temporary columns\n",
    "    nameconv_df = (\n",
    "        nameconv_df\n",
    "        .filter(f.col(\"Queue\").startswith(\"RPA\"))\n",
    "        .withColumn(\"Queue\", f.expr(\"substring(Queue, 1, 7)\"))\n",
    "        .distinct()\n",
    "    )\n",
    "\n",
    "    return nameconv_df"
   ]
